git status показывает статус в текущем каталоге
git log проверка логов событий
git pull проверка изменений на удаленном репо и мержит их с локальными репо
git push origin master мердж изменений, может выдать ошибку, если на сервере были изменения
git add -A добавить все, что есть в репо

git add d:\Labs\Vtornic2018\ForGit\1.txt
git commit

git branch new_test создание новой ветки
git branch проверка статуса веток, активная помечена звездочкой
git checkout new_test переключение между ветками.
git merge [назв. ветки] находясь на ветке с которой нужно слить ветку [назв.ветки] выполнить эту команду для их объединения.
git branch -d [назв.ветки] удалить ветку

git remote -v показывает список удаленных репо с полным адресом
git fetch [имя удал. сервера] команда, что позволяет получить данные с удаленного сервера, которых нет у вас, при этом, 
она не мерджит ваши наработки и новые данные, это нужно будет сделать вручную (в отличие от git pull)
git remote show origin позволяет посмотреть полные данные про удаленный репо и его настройки
git remote rename [старое имя] [новое имя] переименовать сокращение удаленного репо в новый
git remote rm [имя] удаление ссылки на удал. репо

git diff [назв.1]..[назв.2] чтобы увидеть разницу между двумя коммитами
git checkout [commit name] [путь к фалу и его назв.] откатит файл до его состояния на момент указанного коммита 

git revent HEAD откат до предыдущего состояния коммита
git revent [commit name] откат до состояния указанного в коммите
git mergetool граф.интерфейс для решения конфликтов
